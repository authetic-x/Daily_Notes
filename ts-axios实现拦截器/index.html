    axios.interceptors.request.use(config => {
  config.headers[token] = 'xxxxx';
  return config
}, error => {
  return Promise.reject(error);
});

const incep = axios.interceptors.response.use(response => {
  // 对响应数据做一些处理
  return response;
}, error => {
  return Promise.reject(error);
})

axios.interceptors.response.eject(incep);

    拦截器可以添加多个。对于请求拦截器，后添加的先执行；对于响应拦截器，先添加的先执行。拦截器还可以用 eject 方法撤回。

    # TS 实现拦截器

    # 类型定义

    对于请求和响应拦截器，它们都有 use 和 eject 方法，我们可以抽象为一个管理类

    interface ResolvedFn<T> {
  (val: T): T | Promise<T>
}

interface RejectedFn {
  (error: any): any
}

interface InterceptorManager<T> {
  use(resolve: ResolvedFn<T>, rejec?: RejectedFn): number

  eject(id: number): void
}

    # InterceptorManager 具体实现

    interface Interceptor<T> {
  resolve: ResolvedFn<T>
  reject?: RejectedFn
}

class InterceptorManager<T> {
  private interceptors: Array<Interceptor<T> | null>

  constructor() {
    this.interceptors = [];
  }

  use(resolve: ResolvedFn<T>, reject?: RejectedFn) {
    this.interceptors.push({
      resolve,
      reject,
    })
    return this.interceptors.length - 1;
  }

  eject(id: number) {
    if (this.interceptors[id]) {
      this.interceptors[id] = null;
    }
  }

  forEach(fn: (interceptor: Interceptor<T>) => void): void {
    this.interceptors.forEach(interceptor => {
      if (interceptor) {
        fn(interceptor);
      }
    })
  }
}

    use 方法就像 Promise 的 then 方法一样，都是将注册的回调函数保存起来到合适的时机调用。 forEach 提供遍历注册的 interceptors 的接口，接受一个函数参数，可以对每个 interceptor 做处理

    # 请求、响应拦截器实现

    interface Interceptors {
  request: InterceptorManager<AxiosRequestConfig>
  response: InterceptorManager<AxiosResponse>
}

// 兼容dispatchRequest方法
interface PromiseChain {
  resolve: ResolvedFn | ((config: AxiosRequestConfig) => AxiosPromise)
  reject?: RejectedFn
}

class Axios {
  interceptors: Interceptors

  constructor() {
    this.interceptors = {
      request: new InterceptorManager<AxiosRequestConfig>(),
      response: new InterceptorManager<AxiosResponse>(),
    }
  }

  request(url: any, config?: any): AxiosPromise {
    if (typeof url === 'string') {
      if (!config) {
        config = {}
      }
      config.url = url
    } else {
      config = url
    }

    const chain: PromisChain[] = [{
      request: dispatchRequest,
      reject: undefined,
    }]

    this.interceptors.request.forEach(interceptor => {
      chain.unshift(interceptor);
    })

    this.interceptors.response.forEach(interceptor => {
      chain.push(interceptor);
    })

    let promise = Promise.resolve(config);

    while(chain.length) {
      const { resolveFn, rejectFn } = chain.shift()!;
      // resolved函数需要返回config或者response的原因
      promise = promise.then(resolveFn, rejectFn);
    }

    return promise;
  }
}

    其实本质就是将所有注册的拦截器按照约定的顺序调用，但是这里面有异步逻辑，响应拦截器只有等 返回了结果才会被调用，所以需要用到 Promise 的链式调用。
