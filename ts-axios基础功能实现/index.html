    # 封装xhr对象

    function xhr(config: AxiosRequestConfig) {
  const { url, method = 'get', data = null } = config;
  const request = new XMLHttpRequest();

  request.open(method.toUpperCase(), url, true);

  request.send(data);
}

function axios(config: AxiosRequestConfig) {
  xhr(config);
}

    这样已经可以发出一个简单的 ajax 请求了，但是十分简陋，无法接受响应数据，不支持 Promise， 也没有对传入的数据以及发生的错误做处理。

    # 处理请求url参数

    对于请求url参数，我们分对象、数组、Date类型进行处理，还要处理一些边界情况，比如拼接url已有的参数， 对一些特殊字符不进行转义。

    function encode (val: string): string {
  return encodeURIComponent(val)
    .replace(/%40/g, '@')
    .replace(/%3A/gi, ':')
    .replace(/%24/g, '$')
    .replace(/%2C/gi, ',')
    .replace(/%20/g, '+')
    .replace(/%5B/gi, '[')
    .replace(/%5D/gi, ']')
}

function buildURL(url: string, params?: any) {
  if (!params) {
    return url
  }

  const parts: string[] = [];

  Object.keys(params).forEach(key => {
    let val = params[key];
    if (!val) return;

    let values: string[];
    // foo: ['bar', 'baz'] 会被转化为 foo[]=bar&foo[]=baz
    if (Array.isArray(val)) {
      values = val;
      key += '[]'
    } else {
      values = [val];
    }

    values.forEach(val => {
      if (Object.prototype.toString.call(val) === '[object Date]') {
        val = val.toString()
      } else if (Object.prototype.toString.call(val) === '[object Object]') {
        val = JSON.stringify(val);
      }
      parts.push(`${encode(key)}=${encode(val)}`);
    })
  })

  const serializedParams = parts.join('&');

  if (serializedParams) {
    const markIndex = url.indexOf('#');
    if (markIndex !== -1) {
      url = url.slice(0, markIndex);
    }
    url += (url.indexOf('?') !== -1 ? '&' : '?') + serializedParams;
  }
  return url;
}

    # 处理请求header

    对于 data 数据是对象类型，我们将其转换为了 JSON 字符串，但如果 headers 没有配置 Content-Type 属性，服务器就无法正确解析发送的数据，因此我们需要自动设置 Content-Type=application/json;charset=utf-8

    function normalizeHeaderName(headers: any, normalizedName: string) {
  if (headers) {
    Object.keys(headers).forEach(key => {
      if (key !== normalizedName && key.toUpperCase() === normalizedName.toUpperCase()) {
        headers[normalizedName] = headers[key];
        delete headers[key];
      }
    })
  }
}

function processHeaders(headers: any, data: any) {
  normalizeHeaderName(headers, 'Content-Type');

  if (isPlainObject(data)) {
    if (headers && !headers['Content-Type']) {
      headers['Content-Type'] = 'application/json;utf-8';
    }
  }

  return headers;
}

    最后 headers 里的配置项都要用 xhr.setRequestHeader 方法去一一设置。

    # 处理响应数据

    // 定义接口类型
interface AxiosResponse {
  data: any
  status: number
  statusText: string
  headers: any
  config: AxiosRequestConfig
  request: any
}

interface AxiosPromise extends Promise<AxiosResponse> {
}

// 响应返回的原生headers是一段字符串，我们需要手动解析一下
function parseHeaders(headers: string) {
  let parsed = Object.create(null);
  if (!headers) return parsed;

  // windows每行结尾有回车和换行两个字符
  headers.split('\r\n').forEach(line => {
    let [key, val] = line.split(':');
    key = key.trim().toLowerCase();
    if (!key) return;
    if (val) val = val.trim();
    parsed[key] = val;
  });

  return parsed;
}

function xhr(config: AxiosRequestConfig): AxiosPromise {
  return new Promise(resolve => {
    // ...
    request.onreadystatechange = function handleLoad() {
      if (request.readyState !== 4) return;

      const responseHeaders = parseHeaders(request.getAllResponseHeaders());
      const responseData = responseType && responseType !== 'text' ? 
        request.response : request.responseText;
      const response: AxiosResponse = {
        data: responseData,
        status: request.status,
        statusText: request.statusText,
        headers: responseHeaders,
        config,
        request,
      };
      resolve(response);
    }
    // ...
  })
}

// 如果没有正确设置responseType，我们可能需要手动解析返回的JSON字符串
function transformResponse(data: any) {
  if (typeof data === 'string') {
    try {
      data = JSON.parse(data);
    } catch(e) {

  }
  return data;

function axios(config: AxiosRequestConfig) {

    # 错误信息增强

    interface AxiosError extends Error {
  config: AxiosRequestConfig
  code?: string
  request?: any
  response?: AxiosResponse
  isAxiosError: boolean
}

// 鸭子类型
class AxiosError extends Error {
  isAxiosError: boolean
  config: AxiosRequestConfig
  code?: string | null
  request?: any
  response?: AxiosResponse

  constructor(
    message: string,
    config: AxiosRequestConfig,
    code?: string | null,
    request?: any,
    response?: AxiosResponse
  ) {
    super(message)

    this.config = config
    this.code = code
    this.request = request
    this.response = response
    this.isAxiosError = true

    Object.setPrototypeOf(this, AxiosError.prototype)
  }
}

function createError(
  message: string,
  config: AxiosRequestConfig,
  code?: string | null,
  request?: any,
  response?: AxiosResponse
): AxiosError {
  const error = new AxiosError(message, config, code, request, response)

  return error
}

    在构造函数里手动设置实例的原型是为了解决 Typescript 在继承一些内置对象时候的坑，参考。

    为了让错误信息更加规范，在错误处理的地方将 new Error 替换为 createError 即可。
