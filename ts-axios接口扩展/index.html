        # 扩展接口

        axios 除了可以传入 config 对象当作方法使用，还可以调用某个方法 axios.get() 这种形式使用。在 Typescript 里， axios 这种类型被称为混合类型(既是函数又是对象)，对应的可以用 混合接口去描述。

        interface Axios {
  request(config: AxiosRequestConfig): AxiosPromise

  get(url: string, config?: AxiosRequestConfig): AxiosPromise

  // ...
}

interface AxiosInstance extends Axios {
  (config: AxiosRequestConfig): AxiosPromise
}

class Axios {
  request(config: AxiosRequestConfig): AxiosPromise {
    // 这里的dispatch函数其实就是基础功能实现里的axios函数
    return dispatchRequest(config);
  }

  get(url: string, config?: AxiosRequestConfig): AxiosPromise {
    return this._requestWithoutData('get', url, config);
  }

  _requestWithoutData(method: Method, url: string, config?: AxiosRequestConfig) {
    return this.request(
      Object.assign(config || {}, {
        method,
        url,
      });
    );
  }

  // 其它方法实现逻辑类似
}

        到这里已经实现了对 dispatch 函数的封装，接下来只需要将 request 函数和 Axios 类里的 方法混合一下即可。

        function extend<T, U>(to: T, from: U): T & U {
  for (const key in from) {
    (to as T & U)[key] = from[key] as any;
  }
  return to as T & U;
}

function createInstance(): AxiosInstance {
  const context = new Axios();
  // 这一步暂不知有何作用, 因为request里也没有用到this
  const instance = Axios.prototype.request.bind(context);

  extend(instance, context);
  return instance;
}

        # axios函数重载

        axios({
  url: '/api/list',
  method: 'get'
});

axios('/api/login', {
  method: 'post',
  data: {
    username: 'Ethan',
    passwd: '123456',
  }
});

        如果想让 axios 函数有这两种使用形式，我们可以简单重载一下 request 函数

        interface AxiosInstance extends Axios {
  (config: AxiosRequestConfig): AxiosPromise

  (url: string, config?: AxiosRequestConfig): AxiosPromise
}

class Axios {
  request(url: any, config?: AxiosRequestConfig): AxiosPromise {
    if (typeof url === 'string') {
      if (!config) {
        config = {};
      }
      config.url = url;
    } else {
      config = url;
    }
    return dispatchRequest(config);
  }
  //...
}

        # 响应数据支持泛型

        function getUser<T>() {
  return axios.get<ResponseData<T>>('/somepath')
    .then(res => res.data)
    .catch(err => console.log(err));
}

const user = await getUser<User>();

        我们只需要修改一些接口的定义即可。

        interface AxiosResponse<T> {
  data: T
  //...
}

interface AxiosPromise<T = any> extends Promise<AxiosResponse<T>> {
}

// 这里只需要修改对应接口的定义，貌似不需要往Axios类定义中添加泛型
interface Axios {
  request<T = any>(config: AxiosRequestConfig): AxiosPromise<T>
}
